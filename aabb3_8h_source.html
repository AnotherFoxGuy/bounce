24// A min-max representation of a three-dimensional AABB.
25struct b3AABB3
26{
27 b3Vec3 m_lower; // lower vertex
28 b3Vec3 m_upper; // upper vertex
29
30 // Get the support vertex in a given direction.
31 b3Vec3 GetSupportVertex(const b3Vec3& direction) const
32 {
33 b3Vec3 support;
34 support.x = direction.x < 0.0f ? m_lower.x : m_upper.x;
35 support.y = direction.y < 0.0f ? m_lower.y : m_upper.y;
36 support.z = direction.z < 0.0f ? m_lower.z : m_upper.z;
37 return support;
38 }
39
40 // Set this AABB from a list of points.
41 void Set(const b3Vec3* points, u32 count)
42 {
43 m_lower = m_upper = points[0];
44 for (u32 i = 1; i < count; ++i)
45 {
46 m_lower = b3Min(m_lower, points[i]);
47 m_upper = b3Max(m_upper, points[i]);
48 }
49 }
50
51 // Set this AABB from a list of points and a transform.
52 void Set(const b3Vec3* points, u32 count, const b3Transform& xf)
53 {
54 m_lower = m_upper = b3Mul(xf, points[0]);
55 for (u32 i = 1; i < count; ++i)
56 {
57 b3Vec3 v = b3Mul(xf, points[i]);
58 m_lower = b3Min(m_lower, v);
59 m_upper = b3Max(m_upper, v);
60 }
61 }
62
63 // Set this AABB from a triangle.
64 void Set(const b3Vec3& v1, const b3Vec3& v2, const b3Vec3& v3)
65 {
66 m_lower = b3Min(v1, b3Min(v2, v3));
67 m_upper = b3Max(v1, b3Max(v2, v3));
68 }
69
70 // Set this AABB from a center point and a radius vector.
71 void Set(const b3Vec3& center, const b3Vec3& r)
72 {
73 m_lower = center - r;
74 m_upper = center + r;
75 }
76
77 // Set this AABB from a center point and a radius value.
78 void Set(const b3Vec3& center, float32 radius)
79 {
80 b3Vec3 r(radius, radius, radius);
81 Set(center, r);
82 }
83
84 // Extend this AABB by a scalar.
85 void Extend(float32 s)
86 {
87 b3Vec3 r(s, s, s);
88 m_lower -= r;
89 m_upper += r;
90 }
91
92 // Extend this AABB by a radius vector.
93 void Extend(const b3Vec3& r)
94 {
95 m_lower -= r;
96 m_upper += r;
97 }
98
99 // Compute the centroid of this AABB.
100 b3Vec3 Centroid() const
101 {
102 return 0.5f * (m_lower + m_upper);
103 }
104
105 // Compute the width of this AABB.
106 float32 Width() const
107 {
108 return m_upper.x - m_lower.x;
109 }
110
111 // Compute the height of this AABB.
112 float32 Height() const
113 {
114 return m_upper.y - m_lower.y;
115 }
116
117 // Compute the depth of this AABB.
118 float32 Depth() const
119 {
120 return m_upper.z - m_lower.z;
121 }
122
123 // Compute the total of cubic units contained in this AABB.
124 float32 Volume() const
125 {
126 return Width() * Height() * Depth();
127 }
128
129 // Compute the surface area of this AABB.
130 float32 SurfaceArea() const
131 {
132 return 2.0f * (Width() * Depth() + Width() * Height() + Depth() * Height());
133 }
134
135 // Read the index of the longest axis of this AABB.
136 u32 GetLongestAxisIndex() const
137 {
138 b3Vec3 c = Centroid();
139 b3Vec3 r = m_upper - c;
140 float32 max = r[0];
141 u32 i = 0;
142 if (r[1] > max)
143 {
144 max = r[1];
145 i = 1;
146 }
147 if (r[2] > max)
148 {
149 max = r[2];
150 i = 2;
151 }
152 return i;
153 }
154
155 // Test if this AABB contains a point.
156 bool Contains(const b3Vec3& point) const
157 {
158 return m_lower.x <= point.x && point.x <= m_upper.x &&
159 m_lower.y <= point.y && point.y <= m_upper.y &&
160 m_lower.z <= point.z && point.z <= m_upper.z;
161 }
162
163 // Test if this AABB contains another AABB.
164 bool Contains(const b3AABB3& aabb) const
165 {
166 return Contains(aabb.m_lower) && Contains(aabb.m_upper);
167 }

169 // Test if a ray intersects this AABB.
170 bool TestRay(float32& minFraction, const b3Vec3& p1, const b3Vec3& p2, float32 maxFraction) const
171 {
172 b3Vec3 d = p2 - p1;
173
174 float32 lower = 0.0f;
175 float32 upper = maxFraction;
176
177 for (u32 i = 0; i < 3; ++i)
178 {
179 float32 numerators[2], denominators[2];
180
181 numerators[0] = p1[i] - m_lower[i];
182 numerators[1] = m_upper[i] - p1[i];
183
184 denominators[0] = -d[i];
185 denominators[1] = d[i];
186
187 for (u32 j = 0; j < 2; ++j)
188 {
189 float32 numerator = numerators[j];
190 float32 denominator = denominators[j];
191
192 if (denominator == 0.0f)
193 {
194 // s is parallel to this half-space.
195 if (numerator < 0.0f)
196 {
197 // s is outside of this half-space.
198 // dot(n, p1) and dot(n, p2) < 0.
199 return false;
200 }
201 }
202 else
203 {
204 if (denominator < 0.0f)
205 {
206 // s enters this half-space.
207 if (numerator < lower * denominator)
208 {
209 // Increase lower.
210 lower = numerator / denominator;
211 }
212 }
213 else
214 {
215 // s exits the half-space.
216 if (numerator < upper * denominator)
217 {
218 // Decrease upper.
219 upper = numerator / denominator;
220 }
221 }
222 // Exit if intersection becomes empty.
223 if (upper < lower)
224 {
225 return false;
226 }
227 }
228 }
229 }
230
231 B3_ASSERT(lower >= 0.0f && lower <= maxFraction);
232 minFraction = lower;
233 return true;
234 }
235};

237// Compute an AABB that encloses two AABBs.
238inline b3AABB3 b3Combine(const b3AABB3& a, const b3AABB3& b)
239{
240 b3AABB3 aabb;
241 aabb.m_lower = b3Min(a.m_lower, b.m_lower);
242 aabb.m_upper = b3Max(a.m_upper, b.m_upper);
243 return aabb;
244}

246// Test if two AABBs are overlapping.
247inline bool b3TestOverlap(const b3AABB3& a, const b3AABB3& b)
248{
249 return (a.m_lower.x <= b.m_upper.x) && (a.m_lower.y <= b.m_upper.y) && (a.m_lower.z <= b.m_upper.z) &&
250 (a.m_upper.x >= b.m_lower.x) && (a.m_upper.y >= b.m_lower.y) && (a.m_upper.z >= b.m_lower.z);
251}
