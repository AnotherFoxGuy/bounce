43// A plane in constant normal form.
44// dot(n, p) - d = 0.
45struct b3Plane
46{
47 // Does nothing for performance.
48 b3Plane() { }
49
50 // Set this plane from a normal and a signed distance from its origin.
51 b3Plane(const b3Vec3& _normal, float32 _offset)
52 {
53 normal = _normal;
54 offset = _offset;
55 }
56
57 // Set this plane from a normal and a point on the plane.
58 b3Plane(const b3Vec3& _normal, const b3Vec3& _point)
59 {
60 normal = _normal;
61 offset = b3Dot(_normal, _point);
62 }
63
64 // Compute this plane from three non-colinear points.
65 b3Plane(const b3Vec3& A, const b3Vec3& B, const b3Vec3& C)
66 {
67 b3Vec3 N = b3Cross(B - A, C - A);
68 normal = b3Normalize(N);
69 offset = b3Dot(normal, A);
70 }
71
72 b3Vec3 normal;
73 float32 offset;
74};

76// Transform a plane by a given frame.
77inline b3Plane b3Mul(const b3Transform& T, const b3Plane& plane)
78{
79 b3Vec3 normal = b3Mul(T.rotation, plane.normal);

95// Project a point onto a normal plane.
96inline b3Vec3 b3ClosestPointOnPlane(const b3Vec3& P, const b3Plane& plane)
97{
98 float32 fraction = b3Distance(P, plane);

102// Convert a point Q from Cartesian coordinates to Barycentric coordinates (u, v)
103// with respect to a segment AB.
104// The last output value is the divisor.
105inline void b3BarycentricCoordinates(float32 out[3],
106 const b3Vec3& A, const b3Vec3& B,
107 const b3Vec3& Q)
108{
109 b3Vec3 AB = B - A;
110 b3Vec3 QA = A - Q;
111 b3Vec3 QB = B - Q;
112
113 float32 divisor = b3Dot(AB, AB);
114
115 out[0] = b3Dot(QB, AB);
116 out[1] = -b3Dot(QA, AB);
117 out[2] = divisor;
118}
119// Convert a point Q from Cartesian coordinates to Barycentric coordinates (u, v, w)
120// with respect to a triangle ABC.
121// The last output value is the divisor.
122inline void b3BarycentricCoordinates(float32 out[4],
123 const b3Vec3& A, const b3Vec3& B, const b3Vec3& C,
124 const b3Vec3& Q)
125{
126 b3Vec3 AB = B - A;
127 b3Vec3 AC = C - A;
128
129 b3Vec3 QA = A - Q;
130 b3Vec3 QB = B - Q;
131 b3Vec3 QC = C - Q;
132
133 b3Vec3 QB_x_QC = b3Cross(QB, QC);
134 b3Vec3 QC_x_QA = b3Cross(QC, QA);
135 b3Vec3 QA_x_QB = b3Cross(QA, QB);
136
137 b3Vec3 AB_x_AC = b3Cross(AB, AC);
138
139 //float32 divisor = b3Dot(AB_x_AC, AB_x_AC);
140
141 out[0] = b3Dot(QB_x_QC, AB_x_AC);
142 out[1] = b3Dot(QC_x_QA, AB_x_AC);
143 out[2] = b3Dot(QA_x_QB, AB_x_AC);
144 out[3] = out[0] + out[1] + out[2];
145}

147// Convert a point Q from Cartesian coordinates to Barycentric coordinates (u, v, w, x)
148// with respect to a tetrahedron ABCD.
149// The last output value is the (positive) divisor.
150inline void b3BarycentricCoordinates(float32 out[5],
151 const b3Vec3& A, const b3Vec3& B, const b3Vec3& C, const b3Vec3& D,
152 const b3Vec3& Q)
153{
154 b3Vec3 AB = B - A;
155 b3Vec3 AC = C - A;
156 b3Vec3 AD = D - A;
157
158 b3Vec3 QA = A - Q;
159 b3Vec3 QB = B - Q;
160 b3Vec3 QC = C - Q;
161 b3Vec3 QD = D - Q;
162
163 float32 divisor = b3Det(AB, AC, AD);
164 float32 sign = b3Sign(divisor);
165
166 out[0] = sign * b3Det(QB, QC, QD);
167 out[1] = sign * b3Det(QA, QD, QC);
168 out[2] = sign * b3Det(QA, QB, QD);
169 out[3] = sign * b3Det(QA, QC, QB);
170 out[4] = sign * divisor;
171}

173// Project a point onto a segment AB.
174inline b3Vec3 b3ClosestPointOnSegment(const b3Vec3& P, const b3Vec3& A, const b3Vec3& B)
175{
176 float32 wAB[3];
177 b3BarycentricCoordinates(wAB, A, B, P);
178
179 if (wAB[1] <= 0.0f)
180 {
181 return A;
182 }
183
184 if (wAB[0] <= 0.0f)
185 {
186 return B;
187 }
188
189 float32 s = 1.0f / wAB[2];
190 float32 wA = s * wAB[0];
191 float32 wB = s * wAB[1];
192 return wA * A + wB * B;
193}
