80// A cached simplex is used to improve the performance
81// of the GJK when called more than once.
82// Make sure to set cache.count to zero before
83// passing this structure as an argument to GJK when called
84// for the first time.
85struct b3SimplexCache
86{
87 float32 metric; // distance or area or volume
88 u32 iterations; // number of GJK iterations
89 u16 count; // number of support vertices
