26// A transform represents a rigid frame.
27// It has a translation representing a position
28// and a rotation matrix representing an orientation
29// relative to some reference frame.
30struct b3Transform
31{
32 // Default ctor does nothing for performance.
33 b3Transform() { }
34
35 // Set this transform from a rotation matrix and a translation vector.
36 b3Transform(const b3Mat33& _rotation, const b3Vec3& _translation)
37 {
38 rotation = _rotation;
39 position = _translation;
40 }
41
42 // Set this transform from a rotation quaternion and a translation vector.
43 b3Transform(const b3Quat& _rotation, const b3Vec3& _translation)
44 {
45 rotation = b3QuatMat33(_rotation);
46 position = _translation;
47 }
48
49 // Set this transform to the identity transform.
50 void SetIdentity()
51 {
52 rotation.SetIdentity();
53 position.SetZero();
54 }
55
56 b3Mat33 rotation;
57 b3Vec3 position; // in fact a translation
58};

63// Convert a transform to a 4-by-4 transformation matrix.
64inline b3Mat44 b3TransformMat44(const b3Transform& T)

79// Multiply a transform times another transform.
80inline b3Transform b3Mul(const b3Transform& A, const b3Transform& B)
81{
82 // [A y][B x] = [AB Ax+y]
83 // [0 1][0 1] [0 1 ]
84 b3Transform C;
85 C.rotation = b3Mul(A.rotation, B.rotation);
86 C.position = b3Mul(A.rotation, B.position) + A.position;
87 return C;
88}

90// Multiply the transpose of one transform (inverse
91// transform) times another transform (composed transform).
92inline b3Transform b3MulT(const b3Transform& A, const b3Transform& B)
93{
94 //[A^-1 -A^-1*y][B x] = [A^-1*B A^-1(x-y)]
95 //[0 1 ][0 1] [0 1 ]
96 b3Transform C;
97 C.rotation = b3MulT(A.rotation, B.rotation);
98 C.position = b3MulT(A.rotation, B.position - A.position);
99 return C;
100}

102// Multiply the transpose of a transform times a vector.
103// If the transform represents a frame then this transforms
104// the vector from one frame to another (inverse transform).
105inline b3Vec3 b3MulT(const b3Transform& A, const b3Vec3& v)
106{
107 //[A^-1 -A^-1*y][x] = A^-1*x - A^-1*y = A^-1 * (x - y)
108 //[0 1 ][1]
109 return b3MulT(A.rotation, v - A.position);
110}

112// Inverse transform.
113inline b3Transform b3Inverse(const b3Transform& T)
114{
115 b3Transform B;
116 B.rotation = b3Transpose(T.rotation);
117 B.position = b3MulT(T.rotation, -T.position);
118 return B;
119}

204// Motion proxy for TOI computation.
205struct b3Sweep
206{
207 // Get this sweep transform at a given time between [0, 1]
208 b3Transform GetTransform(float32 t) const;
209
210 // Advance to a new initial state.
211 void Advance(float32 t);
212
213 b3Vec3 localCenter; // local center
214
215 b3Quat orientation0; // last orientation
216 b3Vec3 worldCenter0; // last world center
217
218 float32 t0; // last fraction between [0, 1]
219
220 b3Quat orientation; // world orientation
221 b3Vec3 worldCenter; // world center
222};
223
224inline b3Transform b3Sweep::GetTransform(float32 t) const
225{
226 b3Vec3 c = (1.0f - t) * worldCenter0 + t * worldCenter;
227 b3Quat q = (1.0f - t) * orientation0 + t * orientation;
228 q.Normalize();
229
230 b3Transform xf;
231 xf.rotation = b3QuatMat33(q);
232 xf.position = c - b3Mul(q, localCenter);
233 return xf;
234}
235
236inline void b3Sweep::Advance(float32 t)
237{
238 B3_ASSERT(t < 1.0f);
239 float32 dt = (t - t0) / (1.0f / t0);
240 worldCenter += dt * (worldCenter - worldCenter0);
241 orientation += dt * (orientation - orientation0);
242 orientation.Normalize();
243 t0 = t;
244}
