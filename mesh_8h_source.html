24struct b3Triangle
25{
26 // Test if this triangle contains a given vertex.
27 bool TestVertex(u32 v) const
28 {
29 return v == v1 || v == v2 || v == v3;
30 }
31
32 // Test if this triangle contains two vertices.
33 bool TestEdge(u32 _v1, u32 _v2) const
34 {
35 return TestVertex(_v1) && TestVertex(_v2);
36 }
37
38 u32 v1, v2, v3;
39};

78inline b3AABB3 b3Mesh::GetTriangleAABB(u32 index) const
79{
80 const b3Triangle* triangle = triangles + index;
81
82 b3AABB3 aabb;
83 aabb.Set(vertices[triangle->v1], vertices[triangle->v2], vertices[triangle->v3]);

88inline void b3Mesh::BuildTree()
89{
90 b3AABB3* aabbs = (b3AABB3*)b3Alloc(triangleCount * sizeof(b3AABB3));
91 for (u32 i = 0; i < triangleCount; ++i)
92 {
93 aabbs[i] = GetTriangleAABB(i);
94 }
95
96 tree.Build(aabbs, triangleCount);
97
98 b3Free(aabbs);
